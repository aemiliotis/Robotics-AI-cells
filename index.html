import requests
import base64
import json

def call_ai_hub(image_data, lidar_data, voice_cmd=""):
    response = requests.post(
        "https://your-robotics-hub.com/api/process",
        json={
            "cells": ["object_detection", "path_planning", "obstacle_avoidance"],
            "camera_image": base64.b64encode(image_data).decode(),
            "lidar_data": lidar_data,
            "voice_command": voice_cmd,
            "robot_id": "RB-001",
            "timestamp": int(time.time())
        },
        headers={"Authorization": "Bearer YOUR_API_KEY"}
    )
    return response.json()
